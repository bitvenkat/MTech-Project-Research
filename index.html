        function toggleOptimization(strategyId) {

            const detailsElement = document.getElementById('optimization-details');

            const currentTitle = document.getElementById('opt-title').textContent;

            const data = appData.optimizations[strategyId];


            document.querySelectorAll('[id^="btn-"]').forEach(btn => btn.classList.remove('border-teal-500'));

            document.getElementById(`btn-${strategyId}`).classList.add('border-teal-500');


            if (detailsElement.classList.contains('hidden') || currentTitle !== data.title) {

                document.getElementById('opt-title').textContent = data.title;

                document.getElementById('opt-description').textContent = data.description;

                document.getElementById('opt-metrics').textContent = data.metrics;

                document.getElementById('opt-tradeoff').textContent = data.tradeoff;

                detailsElement.classList.remove('hidden');

            } else {

                detailsElement.classList.add('hidden');

            }

        }


        function switchProposal(proposalId) {

            const data = appData.proposals[proposalId];


            document.querySelectorAll('[id^="tab-proposal"]').forEach(btn => {

                btn.classList.remove('bg-teal-700', 'text-white', 'hover:text-teal-700');

                btn.classList.add('bg-white', 'text-gray-600', 'hover:text-teal-700');

            });


            document.getElementById(`tab-proposal${proposalId}`).classList.add('bg-teal-700', 'text-white');

            document.getElementById(`tab-proposal${proposalId}`).classList.remove('text-gray-600', 'hover:text-teal-700');


            document.getElementById('prop-title').textContent = data.title;

            document.getElementById('prop-context').textContent = data.context;

            document.getElementById('prop-novelty').textContent = data.novelty;

            document.getElementById('prop-optimization').textContent = data.optimization;

            document.getElementById('prop-metrics').textContent = data.metrics;

            document.getElementById('prop-methodology').textContent = data.methodology;

        }
